    <form id="predictForm">
        <!-- Example input fields for the features -->
        <input type="text" id="feature1" placeholder="Feature 1">
        <input type="text" id="feature2" placeholder="Feature 2">
        <!-- Add as many feature inputs as required -->
        <button type="submit">Predict</button>
    </form>

    <p id="result"></p>

    <script>
        document.getElementById('predictForm').addEventListener('submit', function(e) {
            e.preventDefault();

            // Collect input data
            let data = {
                feature1: document.getElementById('feature1').value,
                feature2: document.getElementById('feature2').value,
                // Add more features as needed
            };

            // Send POST request to the Flask backend
            fetch('http://127.0.0.1:5000/predict', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(data)
            })
            .then(response => response.json())
            .then(result => {
                // Display the result
                document.getElementById('result').innerText = `Prediction: ${result.prediction}`;
            })
            .catch(error => console.error('Error:', error));
        });
    </script>
